# react-native-pin-lock-view

pin lock view

## Installation

```sh
npm install react-native-pin-lock-view
```

## Usage

```js
import { PinLockView } from 'react-native-pin-lock-view';

// ...

export default function App() {
  return (
    <View style={styles.container}>
      <PinLockView />
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
  },
});
```
